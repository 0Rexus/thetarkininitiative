country_event = { #Otro Wryn Introduced
	id = PEN_formation.2
    title = PEN_formation.2.t
    desc = PEN_formation.2.d
	picture = GFX_report_event_election_vote

	fire_only_once = yes

	trigger = {
		tag = PEN
		NOT = { has_completed_focus = PEN_invite_local_leaders }
		focus_progress = {
			focus = PEN_invite_local_leaders

		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			country_event = { id = PEN_formation.3 days = 7 }
		}
	}
	
	option = {
		name = PEN_formation.2.a
	}
}
