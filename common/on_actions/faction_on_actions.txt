on_actions = {
	on_startup = {
		effect = {
			every_country={
				set_variable = { legitimacy_emp = 0 }
				set_variable = { legitimacy_emp_equilibrium = 0 }
			}
		}
	}
	on_weekly = {
		effect = {
			if = {
				limit = {
					check_variable = { legitimacy_emp_equilibrium < legitimacy_emp }
				}
				subtract_from_variable = { legitimacy_emp = 0.1 }
			}
			if = {
				limit = {
					check_variable = { legitimacy_emp_equilibrium = legitimacy_emp }
				}
			}
			if = {
				limit = {
					check_variable = { legitimacy_emp_equilibrium > legitimacy_emp }
				}
				add_to_variable = { legitimacy_emp = 0.1 }
			}
			clamp_variable = {	
                var = legitimacy_emp
                min = 0
                max = 100
            }
			clamp_variable = {	
                var = legitimacy_emp_equilibrium
                min = 0
                max = 100
            }
		}
	}
}
