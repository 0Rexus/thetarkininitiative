Marketplace_buying_corvettes_republic_menu = {
	context_type = player_context
	window_name = "Marketplace_buying_corvettes_republic_menu"
	visible = {
	   has_country_flag = corvette_republic_market_open
	   has_country_flag = corvette_buying_market_open
	}
	effects = {
		buy_one_price_of_equipment_product_number = {
			if = {
				limit = {
					NOT = {
						has_tech = name_of_equipment_tech
					}
					check_variable = {global.market_ship_holder@price_of_equipment > 0}
				}
				set_technology = {
					name_of_equipment_tech = 1
				}
				add_equipment_production = {
					equipment = {
						type = name_of_equipment_1
					}
					requested_factories = 1
					progress = 1.00
					amount = 1
				}
			}else = {
				add_equipment_production = {
					equipment = {
						type = name_of_equipment_1
					}
					requested_factories = 1
					progress = 1.00
					amount = 1
				}
			}
		}
		auto_buy_price_of_equipment_product_number = {
			set_country_flag = price_of_equipment_auto_buy_enabled
		}
		buy_as_variant_product_number = {
			if = {
				limit = {
					NOT = {
						has_tech = name_of_equipment_tech
					}
					check_variable = {global.market_ship_holder@price_of_equipment > 0}
				}
				set_technology = {
					name_of_equipment_variant_tech = 1
				}
				add_equipment_production = {
					equipment = {
						type = name_of_equipment_1_variant
					}
					requested_factories = 1
					progress = 1.00
					amount = 1
				}
			}else = {
				add_equipment_production = {
					equipment = {
						type = name_of_equipment_1_variant
					}
					requested_factories = 1
					progress = 1.00
					amount = 1
				}
			}
		}
		###AAA###
		## ABOVE COMMENT IS FOR PYTHON SCRIPT DO NOT REMOVE ##
	}
}
