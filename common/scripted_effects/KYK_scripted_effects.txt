###progressive approval 
KYK_progressive_approval_status = {
	if = { limit = { check_variable = {
		var = KYK_progressive_loyalty
		value = 95
		compare = greater_than_or_equals 
	} }
	set_variable = { KYK_progressives_approval = 5 }
	}
	else_if = {
		limit = { check_variable = {
			var = KYK_progressive_loyalty
			value = 75
			compare = greater_than_or_equals 
		} }
		set_variable = { KYK_progressives_approval = 4 }
	}
	else_if = {
		limit = { check_variable = {
			var = KYK_progressive_loyalty
			value = 50
